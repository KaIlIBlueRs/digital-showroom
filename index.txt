<script>
    const { createApp, ref, computed } = Vue;

    createApp({
        setup() {
            // 基础色库数据
            const colors = ref([
                { name: '燕麦暮白', hex: '#F2EFE9', desc: '温柔内敛的色调，为空间注入明亮而解压的轻盈感。' },
                { name: '冷杉青黛', hex: '#2F4F4F', desc: '深邃高贵的森林底色，极具理性与静谧的艺术张力。' },
                { name: '曜石炭黑', hex: '#232526', desc: '纯粹沉稳的极简黑，完美塑造硬朗锋利的现代工业风骨。' },
                { name: '微醺暖驼', hex: '#C5A880', desc: '自带温度的大地色，散发着优雅高档的典雅奢华质感。' },
                { name: '波特红酒', hex: '#722F37', desc: '复古摩登的尊贵红，在戏剧性与内敛间达到精妙平衡。' }
            ]);

            // 表面工艺/纹理数据
            const textures = ref([
                { id: 'matte', name: '微晶哑面', spec: 'Matte Face', desc: '触感如婴儿肌肤般细腻，吸光率高，呈现极其内敛的轻奢哑光视觉，不易留下指纹。' },
                { id: 'brushed', name: '丝理拉丝', spec: 'Brushed', desc: '带有微观的平行金属拉丝肌理，随光线角度变幻折射出冷冽的高科技秩序感与精密感。' },
                { id: 'wood', name: '微刻木纹', spec: 'Natural Grain', desc: '模拟天然年轮与导管的凹凸错落，兼具木质的有机温度与现代复合材料的稳定耐久。' }
            ]);

            // 当前选中的样品配置
            const currentConfig = ref({
                color: colors.value[0],
                texture: textures.value[0]
            });

            // 动态质感美学描述算法
            const activeDescription = computed(() => {
                return `【${currentConfig.value.color.name}】搭配【${currentConfig.value.texture.name}】：${currentConfig.value.color.desc}${currentConfig.value.texture.desc} 这种组合特别适合应用于高端商业空间或前沿住宅的视觉焦点区域。`;
            });

            // AI 助手逻辑
            const aiPrompt = ref('');
            const aiLoading = ref(false);
            const aiResultColor = ref(null);

            // 简易 LLM 接入函数（此结构可以直接兼容任意 OpenAI 标准接口）
            const askAI = async () => {
                if (!aiPrompt.value.trim()) return;
                aiLoading.value = true;
                
                try {
                    /* 
                       💡 生产环境配置说明：
                       你可以直接对接 OpenAI / 智谱AI / DeepSeek 等 API。
                       此处使用 mock 逻辑模拟大模型从自然语言推理出时尚配色的过程。
                    */
                    await new Promise(resolve => setTimeout(resolve, 1200)); // 模拟网络延迟
                    
                    // 模拟大模型根据输入词做语义理解与颜色推导
                    const promptText = aiPrompt.value.toLowerCase();
                    if (promptText.includes('奶油') || promptText.includes('法式')) {
                        aiResultColor.value = { name: 'AI定制：暖樱奶油棕', hex: '#EADBC8', desc: '汲取自浪漫法式烘焙坊，具备高度的包裹感与治愈气息。' };
                    } else if (promptText.includes('科技') || promptText.includes('赛博') || promptText.includes('冷')) {
                        aiResultColor.value = { name: 'AI定制：钛空幻影蓝', hex: '#4A5568', desc: '迎合科技极简趋势，传达理智、克制且前卫的未来氛围。' };
                    } else {
                        // 随机趋势生成
                        const hexes = ['#8E9A86', '#5C3D2E', '#D1B48C', '#483D8B'];
                        const names = ['AI定制：迷雾苔原绿', 'AI定制：深烘浓缩木啡', 'AI定制：威尼斯琥珀', 'AI定制：午夜微光紫'];
                        const idx = Math.floor(Math.random() * hexes.length);
                        aiResultColor.value = { name: names[idx], hex: hexes[idx], desc: '结合了2026年度米兰设计周流行色彩趋势演变而来的高定色。' };
                    }
                } catch (error) {
                    console.error("AI 调色盘服务异常", error);
                } finally {
                    aiLoading.value = false;
                }
            };

            const applyAIColor = () => {
                if (aiResultColor.value) {
                    currentConfig.value.color = { ...aiResultColor.value };
                    aiPrompt.value = '';
                    aiResultColor.value = null;
                }
            };

            // SECTION 2: 现成产品搭配数据（优秀产品展示）
            const products = ref([
                { id: 1, name: '“极光”极简极窄橱柜', tag: '展厅爆款', desc: '全无拉手设计，采用哑面拉丝融合冷色系，适合现代开放式厨房。', color: colors.value[2], texture: textures.value[1], imagePlaceholder: 'kitchen_showcase_01.jpg' },
                { id: 2, name: '“日和”侘寂风茶室背景墙', tag: '高端定制', desc: '大面积温暖色块结合天然微刻木纹，给空间带来松弛舒适的东方禅意。', color: colors.value[0], texture: textures.value[2], imagePlaceholder: 'zen_living_02.jpg' },
                { id: 3, name: '“赫本”复古高级衣帽间', tag: '设计师推荐', desc: '优雅的酒红与高光肌理交织，在柔和射灯下彰显强烈的主人性格与奢华尊贵感。', color: colors.value[4], texture: textures.value[0], imagePlaceholder: 'wardrobe_luxury_03.jpg' }
            ]);

            const loadProductPreset = (product) => {
                currentConfig.value.color = product.color;
                currentConfig.value.texture = product.texture;
                // 滚动回顶部工坊
                document.getElementById('workshop').scrollIntoView({ behavior: 'smooth' });
            };

            return {
                colors,
                textures,
                currentConfig,
                activeDescription,
                aiPrompt,
                aiLoading,
                aiResultColor,
                askAI,
                applyAIColor,
                products,
                loadProductPreset
            };
        }
    }).mount('#app');
</script>
